138        // 以下にjava.lang.Integerのソースコードを抜粋を引用（隘路＝間違いを誘ってしまう元凶）
139
140        /**
141         * Cache to support the object identity semantics of autoboxing for values between
142         * -128 and 127 (inclusive) as required by JLS.
143         *
144         * The cache is initialized on first usage.  The size of the cache
145         * may be controlled by the {@code -XX:AutoBoxCacheMax=<size>} option.
146         * During VM initialization, java.lang.Integer.IntegerCache.high property
147         * may be set and saved in the private system properties in the
148         * sun.misc.VM class.
149         */
150        /*****
151        private static class IntegerCache {
152                static final int low = -128;
153                static final int high;
154                static final Integer cache[];
155
156                static {
157                        // high value may be configured by property
158                        int h = 127;
159                        String integerCacheHighPropValue =
160                        sun.misc.VM.getSavedProperty("java.lang.Integer.IntegerCache.high");
161                        if (integerCacheHighPropValue != null) {
162                                try {
163                                        int i = parseInt(integerCacheHighPropValue);
164                                        i = Math.max(i, 127);
165                                        // Maximum array size is Integer.MAX_VALUE
166                                        h = Math.min(i, Integer.MAX_VALUE - (-low) -1);
167                                } catch( NumberFormatException nfe) {
168                                        // If the property cannot be parsed into an int, ignore it.
169                                }
170                        }
171                        high = h;
172
173                        cache = new Integer[(high - low) + 1];
174                        int j = low;
175                        for(int k = 0; k < cache.length; k++)
176                                cache[k] = new Integer(j++);
177
178                                // range [-128, 127] must be interned (JLS7 5.1.7)
179                                assert IntegerCache.high >= 127;
180                                }
181
182                private IntegerCache() {}
183        }
184        *****/
185
186        /**
187         * Returns an {@code Integer} instance representing the specified
188         * {@code int} value.  If a new {@code Integer} instance is not
189         * required, this method should generally be used in preference to
190         * the constructor {@link #Integer(int)}, as this method is likely
191         * to yield significantly better space and time performance by
192         * caching frequently requested values.
193         *
194         * This method will always cache values in the range -128 to 127,
195         * inclusive, and may cache other values outside of this range.
196         *
197         * @param  i an {@code int} value.
198         * @return an {@code Integer} instance representing {@code i}.
199         * @since  1.5
200         */
201        /*****
202        public static Integer valueOf(int i) {
203                if (i >= IntegerCache.low && i <= IntegerCache.high)
204                        return IntegerCache.cache[i + (-IntegerCache.low)];
205                return new Integer(i);
206        }
