Version 3：描画オブジェクト階層（OpenGLObjectインヘリタンス）を導入する

先のバージョン（Version 2）は、ドラゴンをレンダリングするだけのプログラムに過ぎません。ドラゴン以外のスズメバチ・うさぎ・ペンギン・鬼・赤ちゃんなどの他の立体を描画しようとすると、ドラゴンのプログラムを丸ごとコピーして、他の立体のためにそれぞれ修正を施すことになります。それはコードクローンによるプログラムの量産と呼ばれ、とてもおすすめできない行為です。

もしも、ドラゴンのプログラムの中にバグ（不具合）が見つかった場合、コピーしたコードクローンの中もすべて直さなければなりません。メンテナンス性の低いプログラムの量産、それがコードクローンの行為なのです。オブジェクト指向プログラミングは、このコードクローンを極限まで減少されるために考案された、と申しても過言ではありません。それ（インヘリタンスによる差分プログラミングの手法）を使わない手はないのです。

このバージョン（Version 3）では、描画オブジェクト階層（OpenGLObjectインヘリタンス）を導入して、ドラゴンに限らず、他のスズメバチ・うさぎ・ペンギン・鬼・赤ちゃんなどにも対応できるようにデザインし直して、コードクローンを抑制してしまいましょう。次のクラス構造（インヘリタンスとアグリゲーション：継承と集約）をご覧ください。

ドラゴン（Dragon）は複合描画オブジェクト（OpenGLCompound）のサブクラスになっています。複合描画オブジェクトのスーパクラスは描画オブジェクト（OpenGLObject）です。複合描画オブジェクトの兄弟には三角形オブジェクト（OpenGLTriangle）や多角形オブジェクト（OpenGLPolygon）そしてティーポット（Teapot）がいます。ドラゴンは三角形オブジェクトの複合体なので、複合描画オブジェクトのサブクラスになっているのです。複合描画オブジェクトの内部には、描画オブジェクトのサブクラスのインスタンスであれば、どんなものもcomponentsの中に収納することができます。

以下の差分（変更箇所：編集部分）やクラス階層（インヘリタンス）を参考にしながら、それに続くソースコード（Version 3）と同じになるように、プログラムを編集・加筆してください。mvc4joglという新たなパッケージ（ディレクトリ）を作って、その中に格納することに注意してください。

dragon.Dragon

package dragon;

import com.jogamp.opengl.GL;
import com.jogamp.opengl.GL2;
import com.jogamp.opengl.glu.GLU;
import com.jogamp.opengl.util.gl2.GLUT;

import java.util.Iterator;
import java.util.List;

import mvc4jogl.OpenGLCompound;
import mvc4jogl.OpenGLObject;
import mvc4jogl.OpenGLTriangle;

import utility.StringUtility;

/**
 * ドラゴンをレンダリング（描画）する。
 * XYZ軸に加えてドラゴンを描画する。
 * 視界（視点・注視点・上方向ベクトル・視界角・近・遠）もドラゴンが応答する。
 */
public class Dragon extends OpenGLCompound
{
    /**
     * ドラゴンのコンストラクタである。
     * <a href="http://www.cc.kyoto-su.ac.jp/~atsushi/Programs/VisualWorks/Dragon/dragon.txt">dragon.txt</a>からデータを読み取って三角形群を生成する。
     */
    public Dragon()
    {
        // System.out.println("Dragon");

        String urlString = "http://www.cc.kyoto-su.ac.jp/~atsushi/Programs/VisualWorks/Dragon/dragon.txt";
        List<String> aCollection = StringUtility.readTextFromURL(urlString);
        int numberOfVertexes = 0;
        int numberOfTriangles = 0;
        Iterator<String> anIterator = aCollection.iterator();
        while (anIterator.hasNext())
        {
            String aString = anIterator.next();
            List<String> aList = StringUtility.splitString(aString, " \t\n\r");
            if (aList.size() == 0) { continue; }
            // System.out.println(aList);
            String firstString = aList.get(0);
            if (firstString.compareTo("number_of_vertexes") == 0)
            {
                numberOfVertexes = Integer.valueOf(aList.get(1));
                // System.out.println(numberOfVertexes);
            }
            if (firstString.compareTo("number_of_triangles") == 0)
            {
                numberOfTriangles = Integer.valueOf(aList.get(1));
                // System.out.println(numberOfTriangles);
            }
            if (firstString.compareTo("end_header") == 0)
            {
                double[][] vertexArray = new double[numberOfVertexes][3];
                for (int n = 0; n < numberOfVertexes; n++)
                {
                    aString = anIterator.next();
                    aList = StringUtility.splitString(aString, " \t\n\r");
                    double x = Double.valueOf(aList.get(0));
                    double y = Double.valueOf(aList.get(1));
                    double z = Double.valueOf(aList.get(2));
                    vertexArray[n][0] = x;
                    vertexArray[n][1] = y;
                    vertexArray[n][2] = z;
                    // System.out.printf("%d: %g, %g, %g\n", n, vertexArray[n][0], vertexArray[n][1], vertexArray[n][2]);
                }
                for (int n = 0; n < numberOfTriangles; n++)
                {
                    aString = anIterator.next();
                    aList = StringUtility.splitString(aString, " \t\n\r");
                    int i = Integer.valueOf(aList.get(0)) - 1;
                    int j = Integer.valueOf(aList.get(1)) - 1;
                    int k = Integer.valueOf(aList.get(2)) - 1;
                    double[] vertex1 = vertexArray[i];
                    double[] vertex2 = vertexArray[j];
                    double[] vertex3 = vertexArray[k];
                    OpenGLTriangle aTriangle = new OpenGLTriangle(vertex1, vertex2, vertex3);
                    aTriangle.rgb(0.5d, 0.5d, 1.0d);
                    this.add(aTriangle);
                    // System.out.printf("%d: %s\n", n, aTriangle);
                }
            }
        }

        return;
    }

    /**
     * 視点を応答する。
     * @return 視点座標（x,y,zの配列）
     */
    public double[] eyePoint()
    {
        return new double[] { -5.5852450791872d, 3.07847342734d, 15.794105252496d };
    }

    /**
     * 注視点を応答する。
     * @return 注視点座標（x,y,zの配列）
     */
    public double[] sightPoint()
    {
        return new double[] { 0.27455347776413d, 0.20096999406815d, -0.11261999607086d };
    }

    /**
     * 上方向ベクトルを応答する。
     * @return 上方向ベクトル（x,y,zの配列）
     */
    public double[] upVector()
    {
        return new double[] { 0.1018574904194d, 0.98480906061847d, -0.14062775604137d };
    }

    /**
     * 視野角を応答する。
     * @return 視野角
     */
    public double fovy()
    {
        return 12.642721790235d;
    }

    /**
     * 近を応答する。
     * @return 近
     */
    public double near()
    {
        return 0.01d;
    }

    /**
     * 遠を応答する。
     * @return 遠
     */
    public double far()
    {
        return 100.0d;
    }

    /**
     * レンダリング（描画）する。
     * @param gl グラフィックライブラリ
     * @param glu グラフィックライブラリユーテリティ
     * @param glut グラフィックライブラリユーテリティツールキット
     */
    public void rendering(GL2 gl, GLU glu, GLUT glut)
    {
        double scale = 1.0d;

        gl.glBegin(GL2.GL_LINES);

        // X軸（赤）を描画する。
        gl.glColor4d(1.0d, 0.0d, 0.0d, 1.0d);
        gl.glVertex3d(-1.000d * scale, 0.0d, 0.0d);
        gl.glVertex3d( 1.618d * scale, 0.0d, 0.0d);

        // Y軸（緑）を描画する。
        gl.glColor4d(0.0d, 1.0d, 0.0d, 1.0d);
        gl.glVertex3d(0.0d, -1.000d * scale, 0.0d);
        gl.glVertex3d(0.0d,  1.618d * scale, 0.0d);

        // Z軸（青）を描画する。
        gl.glColor4d(0.0d, 0.0d, 1.0d, 1.0d);
        gl.glVertex3d(0.0d, 0.0d, -1.000d * scale);
        gl.glVertex3d(0.0d, 0.0d,  1.618d * scale);

        gl.glEnd();

        // ドラゴンを描画する。
        super.rendering(gl, glu, glut);

        gl.glFlush();

        return;
    }

    /**
     * ウィンドウのタイトル（ラベル）を応答する。
     * @return タイトル文字列
     */
    public String windowTitle()
    {
        return "Dragon";
    }
}

dragon.Example

package dragon;

import com.jogamp.opengl.GL;
import com.jogamp.opengl.GL2;
import com.jogamp.opengl.glu.GLU;
import com.jogamp.opengl.util.gl2.GLUT;

import com.jogamp.opengl.awt.GLCanvas;
import com.jogamp.opengl.GLAutoDrawable;
import com.jogamp.opengl.GLCapabilities;
import com.jogamp.opengl.GLEventListener;
import com.jogamp.opengl.GLProfile;

import javax.swing.JFrame;
import java.nio.FloatBuffer;

import java.awt.event.WindowAdapter;
import java.awt.event.WindowEvent;

import mvc4jogl.OpenGLObject;

/**
 * JOGL（Java bindings for OpenGL）の例題プログラムである。
 * OpenGLを使ったJavaによる三次元グラフィックスの初歩的な練習になる。
 */
public class Example extends WindowAdapter implements GLEventListener
{
    /**
     * GLオブジェクトを記憶するフィールドである。
     */
    protected GL2 gl;

    /**
     * GLユーティリティを記憶するフィールドである。
     */
    protected GLU glu;

    /**
     * GLユーティリティツールキットを記憶するフィールドである。
     */
    protected GLUT glut;

    /**
     * ビューの幅を記憶するフィールドである。
     */
    protected int width;

    /**
     * ビューの高さを記憶するフィールドである。
     */
    protected int height;

    /**
     * 描画オブジェクトを記憶するフィールドである。
     */
    protected Dragon displayObject;

    /**
     * Exampleのコンストラクタである。
     */
    public Example()
    {
        // System.out.println("Example");

        // ドラゴンのインスタンスを生成する。
        this.displayObject = new Dragon();

        // OpenGL描画のためのキャンバスを生成する。
        GLProfile profile = GLProfile.get(GLProfile.GL2);
        GLCapabilities capabilities = new GLCapabilities(profile);
        capabilities.setDoubleBuffered(true);
        GLCanvas aCanvas = new GLCanvas(capabilities);
        aCanvas.addGLEventListener(this);

        // フレームを生成し、上記のキャンバスを乗せて、ウィンドウとして開く。
        JFrame aFrame = new JFrame(this.displayObject.windowTitle());
        aFrame.add(aCanvas);
        aFrame.addNotify();
        int titleBarHeight = aFrame.getInsets().top;
        aFrame.setSize(400, 400 + titleBarHeight);
        aFrame.setLocationRelativeTo(null);
        // aFrame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
        aFrame.addWindowListener(this);
        aFrame.setFocusable(false);
        aFrame.setVisible(true);

        return;
    }

    /**
     * OpenGL起動時に初期化のために呼び出される。
     * @param aDrawable OpenGLの描画ダーゲット
     */
    public void init(GLAutoDrawable aDrawable)
    {
        // System.out.println("init");

        // グラフィックライブラリ（gl）を獲得し、
        // グラフィックライブラリユーティリティ（glu）と
        // グラフィックライブラリユーティリティツールキット（glut）を生成する。
        this.gl = aDrawable.getGL().getGL2();
        this.glu = new GLU();
        this.glut = new GLUT();

        // 材質の有効化
        // 陰線隠面処理の有効化
        // 片面表示機能の無効化
        // 法線ベクトルの有効化
        this.gl.glEnable(GL2.GL_COLOR_MATERIAL);
        this.gl.glEnable(GL2.GL_DEPTH_TEST);
        this.gl.glDisable(GL2.GL_CULL_FACE);
        this.gl.glEnable(GL2.GL_NORMALIZE);

        return;
    }

    /**
     * OpenGL描画を行う際に呼び出される。
     * @param aDrawable OpenGLの描画ダーゲット
     */
    public void display(GLAutoDrawable aDrawable)
    {
        // System.out.println("display");

        // 視界（視点・注視点・上方向ベクトル・視界角・近・遠）をドラゴンから取得して整（設）える。
        double[] eyePoint = this.displayObject.eyePoint();
        double[] sightPoint = this.displayObject.sightPoint();
        double[] upVector = this.displayObject.upVector();
        double fovy = this.displayObject.fovy();
        double near = this.displayObject.near();
        double far = this.displayObject.far();

        // キャンバス（ウィンドウ）の大きさからアスペクト比を求める。
        double aspect = (double)width / (double)height;

        // プロジェクション（遠近法）を設定する。
        this.gl.glMatrixMode(GL2.GL_PROJECTION);
        this.gl.glLoadIdentity();
        this.glu.gluPerspective(fovy, aspect, near, far);

        // 視界を設定する。
        this.gl.glMatrixMode(GL2.GL_MODELVIEW);
        this.gl.glLoadIdentity();
        this.glu.gluLookAt(
                           eyePoint[0], eyePoint[1], eyePoint[2],
                           sightPoint[0], sightPoint[1], sightPoint[2],

 */
public class OpenGLCompound extends OpenGLObject
{
    /**
     * 描画オブジェクト群を記憶するフィールドである。
     */
    protected ArrayList<OpenGLObject> components;

    /**
     * 複合形のコンストラクタである。
     */
    public OpenGLCompound()
    {
        super();

        this.components = new ArrayList<OpenGLObject>();

        return;
    }

    /**
     * 描画オブジェクトを複合形の中に追加する。
     * @param aBody 描画オブジェクト
     */
    public void add(OpenGLObject aBody)
    {
        this.components.add(aBody);

        return;
    }

    /**
     * 複合形の構成要素列に応答する。
     * @return 複合形の構成要素列（描画オブジェクト列）
     */
    public ArrayList<OpenGLObject> components()
    {
        return this.components;
    }

    /**
     * レンダリング（描画）する。
     * @param gl グラフィックライブラリ
     * @param glu グラフィックライブラリユーテリティ
     * @param glut グラフィックライブラリユーテリティツールキット
     */
    public void rendering(GL2 gl, GLU glu, GLUT glut)
    {
        super.rendering(gl, glu, glut);

        for (OpenGLObject each: components)
        {
            each.rendering(gl, glu, glut);
        }
        gl.glFlush();

        return;
    }
}

mvc4jogl.OpenGLPolygon

package mvc4jogl;

import com.jogamp.opengl.GL;
import com.jogamp.opengl.GL2;
import com.jogamp.opengl.glu.GLU;
import com.jogamp.opengl.util.gl2.GLUT;

/**
 * OpenGLの多角形（描画オブジェクト）である。
 */
public class OpenGLPolygon extends OpenGLObject
{
    /**
     * 頂点列を記憶するフィールドである。
     */
    protected double[][] vertexes;

    /**
     * 法線単位ベクトルを記憶するフィールドである。
     */
    protected double[] normalUnitVector;

    /**
     * 多角形のコンストラクタである。
     * @param vertexArray 座標（doubleの配列）の配列
     */
    public OpenGLPolygon(double[][] vertexArray)
    {
        super();

        this.vertexes = vertexArray;
        this.normalUnitVector = new double[3];

        double x = 0.0d;
        double y = 0.0d;
        double z = 0.0d;
        int length = this.vertexes.length;
        for (int i = 0; i < length; i++)
        {
            int j = (i + 1) % length;
            int k = (i + 2) % length;
            double ux = this.vertexes[j][0] - this.vertexes[i][0];
            double uy = this.vertexes[j][1] - this.vertexes[i][1];
            double uz = this.vertexes[j][2] - this.vertexes[i][2];
            double vx = this.vertexes[k][0] - this.vertexes[j][0];
            double vy = this.vertexes[k][1] - this.vertexes[j][1];
            double vz = this.vertexes[k][2] - this.vertexes[j][2];
            x = x + ((uy * vz) - (uz * vy));
            y = y + ((uz * vx) - (ux * vz));
            z = z + ((ux * vy) - (uy * vx));
        }
        double distance = Math.sqrt((x * x) + (y * y) + (z * z));
        this.normalUnitVector[0] = x / distance;
        this.normalUnitVector[1] = y / distance;
        this.normalUnitVector[2] = z / distance;

        return;
    }

    /**
     * レンダリング（描画）する。
     * @param gl グラフィックライブラリ
     * @param glu グラフィックライブラリユーテリティ
     * @param glut グラフィックライブラリユーテリティツールキット
     */
    public void rendering(GL2 gl, GLU glu, GLUT glut)
    {
        super.rendering(gl, glu, glut);

        gl.glBegin(GL2.GL_POLYGON);
        gl.glNormal3dv(OpenGLObject.asDoubleBuffer(this.normalUnitVector));
        int length = this.vertexes.length;
        for (int i = 0; i < length; i++)
        {
            gl.glVertex3dv(OpenGLObject.asDoubleBuffer(this.vertexes[i]));
        }
        gl.glEnd();
        gl.glFlush();

        return;
    }
}

package mvc4jogl;

import com.jogamp.opengl.GL2;
import com.jogamp.opengl.glu.GLU;
import com.jogamp.opengl.util.gl2.GLUT;

/**
 * OpenGLの三角形（描画オブジェクト）である。
 */
public class OpenGLTriangle extends OpenGLObject
{
    /**
     * 第1頂点を記憶するフィールドである。
     */
    protected double[] vertex1;

    /**
     * 第2頂点を記憶するフィールドである。
     */
    protected double[] vertex2;

    /**
     * 第3頂点を記憶するフィールドである。
     */
    protected double[] vertex3;

    /**
     * 法線単位ベクトルを記憶するフィールドである。
     */
    protected double[] normalUnitVector;

    /**
     * 三角形のコンストラクタである。
     * @param aVertex1 第1頂点
     * @param aVertex2 第2頂点
     * @param aVertex3 第3頂点
     */
    public OpenGLTriangle(double[] aVertex1, double[] aVertex2, double[] aVertex3)
    {
        super();

        this.vertex1 = aVertex1;
        this.vertex2 = aVertex2;
        this.vertex3 = aVertex3;
        this.normalUnitVector = new double[3];

        double ux = this.vertex2[0] - this.vertex1[0];
        double uy = this.vertex2[1] - this.vertex1[1];
        double uz = this.vertex2[2] - this.vertex1[2];
        double vx = this.vertex3[0] - this.vertex1[0];
        double vy = this.vertex3[1] - this.vertex1[1];
        double vz = this.vertex3[2] - this.vertex1[2];
        double x = (uy * vz) - (uz * vy);
        double y = (uz * vx) - (ux * vz);
        double z = (ux * vy) - (uy * vx);
        double distance = Math.sqrt((x * x) + (y * y) + (z * z));
        this.normalUnitVector[0] = x / distance;
        this.normalUnitVector[1] = y / distance;
        this.normalUnitVector[2] = z / distance;

        return;
    }

    /**
     * レンダリング（描画）する。
     * @param gl グラフィックライブラリ
     * @param glu グラフィックライブラリユーテリティ
     * @param glut グラフィックライブラリユーテリティツールキット
     */
    public void rendering(GL2 gl, GLU glu, GLUT glut)
    {
        super.rendering(gl, glu, glut);

        gl.glBegin(GL2.GL_TRIANGLES);
        gl.glNormal3dv(OpenGLObject.asDoubleBuffer(this.normalUnitVector));
        gl.glVertex3dv(OpenGLObject.asDoubleBuffer(this.vertex1));
        gl.glVertex3dv(OpenGLObject.asDoubleBuffer(this.vertex2));
        gl.glVertex3dv(OpenGLObject.asDoubleBuffer(this.vertex3));
        gl.glEnd();
        gl.glFlush();

        return;
    }
